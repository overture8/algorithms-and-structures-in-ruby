    def is_palindrome?(string)
  string == string.reverse
end

def is_palindrome_b?(string)
  if string.length == 1 || string.length == 0
    true
  else
    if string[0] == string[-1]
      is_palindrome_b?(string[1..-2])
    else
      false
    end
  end
end
